## JavaScript

```js
/**
 * @param {number} n
 */
function fibonacci(n) {
  if (n === 1) return 1;
  if (n === 2) return 2;
  return fibonacci(n - 1) + fibonacci(n - 2);
}
```

## Diff JavaScript

```diff-js
@@ -4,6 +4,5 @@
- let foo = bar.baz([1, 2, 3]);
- foo = foo + 1;
+ const foo = bar.baz([1, 2, 3]) + 1;
  console.log(`foo: ${foo}`);
```
